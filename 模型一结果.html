def calcCosDist(user1,user2):
    sum_x=0.0
    sum_y=0.0
    sum_xy=0.0
    for key1 in user1:
        for key2 in user2:
            if key1[0]==key2[0] :
                sum_xy+=key1[1]*key2[1]
                sum_y+=key2[1]*key2[1]
                sum_x+=key1[1]*key1[1]

    if sum_xy == 0.0 :
        return 0
    sx_sy=math.sqrt(sum_x*sum_y) 
    return sum_xy/sx_sy

#   输出:与指定用户最相邻的邻居列表
def calcNearestNeighbor(userid,users_dic,item_dic):
    neighbors=[]
    for item in users_dic[userid]:
        for neighbor in item_dic[item[0]]:
            if neighbor != userid and neighbor not in neighbors: 
                neighbors.append(neighbor)
    neighbors_dist=[]
    for neighbor in neighbors:
        dist=calcCosDist(users_dic[userid],users_dic[neighbor])  #calcSimlaryCosDist  calcCosDist calcCosDistSpe
        neighbors_dist.append([dist,neighbor])
    neighbors_dist.sort(reverse=True)
    return  neighbors_dist

def ScorePred(userid):
    rec565=calcNearestNeighbor(userid,user_rate_dic,item_to_user)
    neighbors=rec565#输出形式是[相似度，用户id]
    recommend_dic={}
    #找出邻居id，输出根据邻居的推荐物品
    for neighbor in neighbors:
        neighbor_user_id=neighbor[1]
        products=user_rate_dic[neighbor_user_id]
        for prd in products:
            if prd[0] not in recommend_dic:
                recommend_dic[prd[0]]=neighbor[0]
            else:
                recommend_dic[prd[0]]+=neighbor[0]


    #建立推荐列表
    recommend_list=[]
    for key in recommend_dic:
        recommend_list.append([recommend_dic[key],key])

    recommend_list.sort(reverse=True)


    #提取出相似度高且可推荐的用户id
    prod_rec_user=[]
    reclist=[]
    import operator
    for i in rec565:
        if len(set(id_prodhold[i[1]])-set(id_prodhold[userid]))==1:
            if len(set(id_prodhold[i[1]])&set(id_prodhold[userid]))==len(set(id_prodhold[userid])):
                reclist.append(i)
    for i in reclist:
        for j in reclist :
            if i!=j:
                if operator.eq(id_prodhold[i[1]],id_prodhold[j[1]])==True:
                    prod_rec_user.append(i)
                    prod_rec_user.append(j)
    prod_rec_user2=prod_rec_user[0:2]

    SelectScore=scaled_Product.iloc[[userid,prod_rec_user2[0][1],prod_rec_user2[1][1]],]

    avg2=sum(SelectScore.iloc[0,:])/Prod_num[userid]
    avgx=sum(SelectScore.iloc[1,:])/Prod_num[prod_rec_user2[0][1]]
    avgy=sum(SelectScore.iloc[2,:])/Prod_num[prod_rec_user2[1][1]]

    global n
    global score2
    n=set(id_prodhold[prod_rec_user2[0][1]])-set(id_prodhold[userid])#预测物品的序号
    score2=avg2+(prod_rec_user2[0][0]*(SelectScore.iloc[1,sum(n)]-avgx)+prod_rec_user2[1][0]*(SelectScore.iloc[2,sum(n)]-avgy))/(SelectScore.iloc[1,sum(n)]+SelectScore.iloc[2,sum(n)])

    return n,score2,prod_rec_user2
